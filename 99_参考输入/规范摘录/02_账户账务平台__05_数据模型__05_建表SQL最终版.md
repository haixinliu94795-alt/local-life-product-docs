# 建表 SQL 最终版

> 版本：V6.4  
> 状态：stable  
> 用途：账户账务平台一阶段开发建表依据  
> 说明：本 SQL 面向一阶段重建场景，建表前先执行 `DROP TABLE IF EXISTS`。生产环境已有数据时不得直接执行全量重建 SQL，必须另行编写迁移脚本。
>
> 一阶段模板为代码内置能力，`accounting_request` 不落 `template_code/template_version` 等模板治理字段；如需追溯本次请求依据，可通过 `caller_system / biz_domain / accounting_scene / biz_no / event_type / request_snapshot` 和对应流水还原。

## 1. fund_subject

```sql
DROP TABLE IF EXISTS `fund_subject`;

CREATE TABLE `fund_subject` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `subject_no` varchar(64) NOT NULL COMMENT '资金主体编号，账户账务内部编号',
  `identity_id` varchar(64) NOT NULL COMMENT '统一身份 / 业务主体ID',
  `identity_role` varchar(32) NOT NULL COMMENT '身份角色，如 MERCHANT/PROMOTER/SALESMAN/GROUP_MEAL_LEADER/PLATFORM',
  `identity_name` varchar(128) NOT NULL COMMENT '身份名称快照',
  `status` varchar(32) NOT NULL DEFAULT 'NORMAL' COMMENT '主体状态',
  `remark` varchar(256) DEFAULT NULL COMMENT '备注',
  `create_time` datetime(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) COMMENT '创建时间',
  `update_time` datetime(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3) COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_fund_subject_no` (`subject_no`),
  UNIQUE KEY `uk_fund_subject_identity_role` (`identity_id`, `identity_role`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='资金主体表';
```

## 2. fund_account

```sql
DROP TABLE IF EXISTS `fund_account`;

CREATE TABLE `fund_account` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `account_no` varchar(64) NOT NULL COMMENT '资金账户编号，账户账务内部编号',
  `subject_no` varchar(64) NOT NULL COMMENT '资金主体编号，冗余用于查询和排查',
  `identity_id` varchar(64) NOT NULL COMMENT '统一身份 / 业务主体ID快照',
  `identity_role` varchar(32) NOT NULL COMMENT '身份角色快照',
  `identity_name` varchar(128) DEFAULT NULL COMMENT '身份名称快照',
  `account_type` varchar(32) NOT NULL COMMENT '账户类型：SETTLEMENT/COMMISSION/PLATFORM_FUND',
  `account_nature` varchar(32) NOT NULL COMMENT '账户性质：FUND_POSITION/PAYABLE_CLAIM',
  `available_amount` bigint NOT NULL DEFAULT 0 COMMENT '可用余额，单位分，人民币',
  `frozen_amount` bigint NOT NULL DEFAULT 0 COMMENT '冻结余额，单位分，人民币',
  `status` varchar(32) NOT NULL DEFAULT 'NORMAL' COMMENT '账户状态',
  `version` bigint unsigned NOT NULL DEFAULT 0 COMMENT '乐观锁版本号',
  `create_time` datetime(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) COMMENT '创建时间',
  `update_time` datetime(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3) COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_fund_account_no` (`account_no`),
  UNIQUE KEY `uk_fund_account_identity_role_type` (`identity_id`, `identity_role`, `account_type`),
  KEY `idx_fund_account_subject_no` (`subject_no`),
  KEY `idx_fund_account_identity_role` (`identity_id`, `identity_role`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='资金账户表';
```

## 3. accounting_request

```sql
DROP TABLE IF EXISTS `accounting_request`;

CREATE TABLE `accounting_request` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `accounting_no` varchar(64) NOT NULL COMMENT '账务处理编号',
  `caller_system` varchar(64) NOT NULL COMMENT '调用方系统，表示本次账务请求由哪个后端系统发起',
  `biz_domain` varchar(64) NOT NULL COMMENT '业务归属域，表示biz_no由哪个业务域解释和管理',
  `accounting_scene` varchar(64) NOT NULL COMMENT '标准账务场景，用于模板匹配和报表分类',
  `biz_no` varchar(64) NOT NULL COMMENT '业务单号',
  `event_type` varchar(64) NOT NULL COMMENT '事件类型',
  `idempotent_key` varchar(128) NOT NULL COMMENT '幂等键',
  `request_hash` varchar(64) NOT NULL COMMENT '请求核心参数指纹',
  `amount` bigint NOT NULL COMMENT '本次业务账务金额，单位分，人民币',
  `status` varchar(32) NOT NULL COMMENT '处理状态',
  `request_snapshot` json DEFAULT NULL COMMENT '请求脱敏快照',
  `error_code` varchar(64) DEFAULT NULL COMMENT '错误码',
  `error_message` varchar(256) DEFAULT NULL COMMENT '错误信息',
  `create_time` datetime(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) COMMENT '创建时间',
  `update_time` datetime(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3) COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_accounting_no` (`accounting_no`),
  UNIQUE KEY `uk_accounting_idempotent` (`biz_domain`, `idempotent_key`),
  KEY `idx_accounting_biz` (`biz_domain`, `biz_no`, `accounting_scene`),
  KEY `idx_accounting_caller_time` (`caller_system`, `create_time`),
  KEY `idx_accounting_status_time` (`status`, `create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='账务请求表';
```

## 4. fund_account_flow

```sql
DROP TABLE IF EXISTS `fund_account_flow`;

CREATE TABLE `fund_account_flow` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `flow_no` varchar(64) NOT NULL COMMENT '账户流水编号',
  `accounting_no` varchar(64) NOT NULL COMMENT '账务处理编号',
  `flow_seq` int unsigned NOT NULL COMMENT '同一账务处理内的流水顺序，从1开始',
  `account_no` varchar(64) NOT NULL COMMENT '资金账户编号',
  `subject_no` varchar(64) NOT NULL COMMENT '资金主体编号',
  `identity_id` varchar(64) NOT NULL COMMENT '统一身份 / 业务主体ID快照',
  `identity_role` varchar(32) NOT NULL COMMENT '身份角色快照',
  `account_type` varchar(32) NOT NULL COMMENT '账户类型快照',
  `flow_direction` varchar(32) NOT NULL COMMENT '流水方向：INCREASE/DECREASE/FREEZE/DEDUCT_FROZEN/UNFREEZE',
  `amount` bigint NOT NULL COMMENT '变更金额，单位分，人民币',
  `available_before` bigint NOT NULL COMMENT '变更前可用余额，单位分，人民币',
  `available_after` bigint NOT NULL COMMENT '变更后可用余额，单位分，人民币',
  `frozen_before` bigint NOT NULL COMMENT '变更前冻结余额，单位分，人民币',
  `frozen_after` bigint NOT NULL COMMENT '变更后冻结余额，单位分，人民币',
  `caller_system` varchar(64) NOT NULL COMMENT '调用方系统，与账务请求保持一致，便于流水独立追溯',
  `biz_domain` varchar(64) NOT NULL COMMENT '业务归属域，与账务请求保持一致，便于按业务域查询流水',
  `accounting_scene` varchar(64) NOT NULL COMMENT '标准账务场景，用于模板匹配和报表分类',
  `biz_no` varchar(64) NOT NULL COMMENT '业务单号',
  `event_type` varchar(64) NOT NULL COMMENT '事件类型',
  `remark` varchar(256) DEFAULT NULL COMMENT '备注',
  `create_time` datetime(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) COMMENT '创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_flow_no` (`flow_no`),
  UNIQUE KEY `uk_flow_accounting_seq` (`accounting_no`, `flow_seq`),
  KEY `idx_flow_account_time` (`account_no`, `create_time`),
  KEY `idx_flow_accounting_no` (`accounting_no`),
  KEY `idx_flow_biz` (`biz_domain`, `biz_no`, `accounting_scene`),
  KEY `idx_flow_caller_time` (`caller_system`, `create_time`),
  KEY `idx_flow_identity_role_time` (`identity_id`, `identity_role`, `create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='账户流水表';
```
